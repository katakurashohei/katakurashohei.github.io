 4  (:use #:cl #:vecto))

 6(in-package #:hilbert-curve)

 8(defparameter *dx* #(1 0 -1 0))
 9(defparameter *dy* #(0 1 0 -1))
10
11(defun l-system (order axiom rules)
12  (dotimes (_ order)
13    (let ((next ""))
14      (loop for c across axiom do
15        (let ((replacement (assoc c rules)))
16          (setf next (concatenate 'string next
17                                  (if replacement
18                                      (cdr replacement)
19                                      (string c))))))
20      (setf axiom next)))
21  axiom)
22
23(defun lindenmayer-hilbert (order)
24  (let ((axiom "A")
25        (rules (list (cons #\A "-BF+AFA+FB-")
26                     (cons #\B "+AF-BFB-FA+"))))
27    (l-system order axiom rules)))
28
29(defun turtle-drawing (size order)
30  (let* ((instr (coerce order 'list))
31         (step (/ size (expt 2 order)))
32         (x 0)
33         (y 0)
34         (dir 0)
35         (points (list (list x y))))
36    (dolist (c instr (nreverse points))
37      (case c
38        (#\+ (setf dir (mod (+ dir 1) 4)))
39        (#\- (setf dir (mod (+ dir 3) 4)))
40        (#\F
41         (incf x (aref *dx* dir))
42         (incf y (aref *dy* dir))
43         (push (list (* x step) (* y step)) points))))))
44
45(defun draw-hilbert-png (size order file)
46  (with-canvas (:width size :height size)
47    (let ((points (turtle-drawing size (lindenmayer-hilbert order))))
48      (stroke)
49      (translate 0 (- size 1))
50      (move-to 0 0)
51      (mapcar (lambda (pt) (apply #'line-to pt))points)
52      (stroke)
53      (save-png file))))
